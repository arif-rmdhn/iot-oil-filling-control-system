//------------------------------------------------------- FUNCTION VOID LOOP --------------------------------------------------//
void loop() {
  while (WiFi.status() != WL_CONNECTED) {
    digitalWrite(relay_pump, 0);
    WiFi.begin(ssid, pass);
    lcd.clear();
    lcd.setCursor(2, 0);
    lcd.print("RECONNECTING");
    lcd.setCursor(4, 1);
    lcd.print("TO WIFI!");
    delay(500);
    lcd.clear();
    delay(500);
  }

  Blynk.run();
  read_keyPad();
  read_sensor();

  lcd.clear();
  lcd.setCursor(0, 0);
  lcd.print("Volume:" + String(data_prev_sen) + "ml");
  cek_kondisi();
}


//------------------------------------------------------- FUNCTION READ SENSOR --------------------------------------------------//
void read_sensor() {
  while (readIndex < num_index) {
    read_keyPad();
    digitalWrite(TRIG_PIN, HIGH);
    delayMicroseconds(10);
    digitalWrite(TRIG_PIN, LOW);

    duration_us = pulseIn(ECHO_PIN, HIGH);

    /* Tanpa Menggunakan Regresi Liniear */
    // distance_cm = 0.0343 * duration_us / 2;

    /* Menggunakan Regresi Liniear */
    distance_cm = slope * duration_us + intercept;

    get_average_data[readIndex] = distance_cm;

    readIndex += 1;
    delay(40);
  }

  for (int i = 0; i < num_index; i++) {
    data1 += get_average_data[i];
  }
  data1 = data1 / readIndex;          // Nilai Rata-Rata
  realData1 = 14.5 - data1;           // Konversi ke CM
  realData2 = realData1 * 9.3 * 9.3;  // Konversi ml

  if (data_prev_sen != realData2) {
    data_prev_sen = realData2;
    data_prev_sen = constrain(data_prev_sen, 0, 1200);
    dataJarak(data_prev_sen);
  }

  readIndex = 0;
  data1 = 0;
  data2 = 0;

  for (int i = 0; i < num_index; i++) {
    get_average_data[i] = 0;
  }
}

//------------------------------------------------------- FUNCTION SCEK KONDISI --------------------------------------------------//
void cek_kondisi() {
  switch (condition) {
    case 1:
      cek_level_1();
      break;
    case 2:
      cek_level_2();
      break;
  }
  pump_stat(status_motor);
}

//------------------------------------------------------- FUNCTION CEK LEVEL 1 --------------------------------------------------//
void cek_level_1() {

  if (data_prev_sen < set_point_high && !interupsi) {
    status_motor = 1;
  } else if (data_prev_sen >= set_point_high && !interupsi) {
    status_motor = 0;
  } else {
    status_motor = 0;
  }

  digitalWrite(relay_pump, status_motor);
  lcd.setCursor(0, 1);
  lcd.print("D:Set|SP:" + String(set_point_high) + "ml");
}

//------------------------------------------------------- FUNCTION CEK LEVEL 2 --------------------------------------------------//
void cek_level_2() {

  if ((data_prev_sen <= set_point_low) && (!status_motor) && (!interupsi)) {
    status_motor = 1;
  } else if ((data_prev_sen >= set_point_high) && (status_motor) && (!interupsi)) {
    status_motor = 0;
  } else {
    status_motor = 0;
  }

  digitalWrite(relay_pump, status_motor);
  lcd.setCursor(0, 1);
  lcd.print("HIGH:" + String(set_point_high) + "|LOW:" + String(set_point_low));
}


//------------------------------------------------------- FUNCTION READ KEYPAD --------------------------------------------------//
void read_keyPad() {
  data_Keypad = customKeypad.getKey();
  if (data_Keypad) {
    if (data_Keypad == 'D') {
      Setting_Mode();
    }
    if (data_Keypad == 'A') {
      data_Keypad = ' ';
      stat_mode_1(1);
      stat_mode_2(0);
      lcd.clear();
      lcd.setCursor(0, 0);
      lcd.print("Masuk ke Mode 1");
      delay(2000);
      lcd.clear();
      condition = 1;
    }
    if (data_Keypad == 'B') {
      data_Keypad = ' ';
      stat_mode_1(0);
      stat_mode_2(1);
      lcd.clear();
      lcd.setCursor(0, 0);
      lcd.print("Masuk ke Mode 2");
      delay(2000);
      lcd.clear();
      condition = 2;
    }
    if (data_Keypad == 'C' && !interupsi) {
      lcd.clear();
      lcd.setCursor(6, 0);
      lcd.print("STATUS");
      lcd.setCursor(0, 1);
      lcd.print("Pump Non Active");
      delay(2000);
      lcd.clear();
      data_Keypad = ' ';
      interupsi = 1;
    }
    if (data_Keypad == 'C' && interupsi) {
      lcd.clear();
      lcd.setCursor(6, 0);
      lcd.print("STATUS");
      lcd.setCursor(1, 1);
      lcd.print("Pump is Active");
      delay(2000);
      lcd.clear();
      data_Keypad = ' ';
      interupsi = 0;
    }
    if (data_Keypad == '#') {
      informasi_button();
    }
    if (data_Keypad == '*' && !status_pump2) {
      data_Keypad = ' ';
      status_pump2 = 1;
      digitalWrite(relay_pump2, status_pump2);
    }
    if (data_Keypad == '*' && status_pump2) {
      data_Keypad = ' ';
      status_pump2 = 0;
      digitalWrite(relay_pump2, status_pump2);
    }
  }
}

  lcd.clear();

  for (int i = 0; i <= 7; i++) {
    if (i % 2 == 0) {

      lcd.print("                     ");
    } else {
      if (i == 1) {
        lcd.setCursor(0, 1);

        delay(1500);
      } else if (i == 3) {
        lcd.setCursor(0, 1);

        delay(1500);
      } else if (i == 5) {
        lcd.setCursor(0, 1);
        lcd.print("B: Mode 2 SP");
        delay(1500);
      } else {
        lcd.setCursor(0, 1);
        lcd.print("C: PUMP ON & OFF");
        delay(1500);
      }
    }
  }
}
